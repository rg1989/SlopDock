    6. Add useEffect for SSE subscription to /api/canvas/events:
       ```typescript
       useEffect(() => {
         if (!cwd) return;
         const es = new EventSource('/api/canvas/events');
         es.onmessage = (e) => {
           const data = JSON.parse(e.data);
           setCanvasTabs(data.tabs ?? []);
           // If activeCanvasId no longer in tabs, fall back to first tab
           setActiveCanvasId(prev =>
             data.tabs?.find((t: CanvasTab) => t.id === prev) ? prev : data.tabs?.[0]?.id ?? null
           );
         };
         return () => es.close();
       }, [cwd]);
       ```
